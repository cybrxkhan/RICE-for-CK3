# Johannes Scotus Erigena
# Irish Neoplatonist philosopher and theologian
# Considered one of the most gifted Christian thinkers of the 9th century
# Useful sources besides Wikipedia:
# https://plato.stanford.edu/entries/scottus-eriugena/
# https://www.dib.ie/biography/john-scottus-eriugena-a4287
RICE_irish_003 = {
	name = "Johannes"
	dynasty = RICE_dynasty_eriugena
	religion = "insular_celtic"
	culture = "irish"
	# According to his contemporaries, he was very, very good at logical reasoning when debating with others; he was also patronized by French rulers
	diplomacy = 8
	martial = 2
	stewardship = 6 	# Apparently there's evidence he had good expertise in Irish law of the time
	intrigue = 7 		# Said to be well-versed in political affairs
	learning = 8
	trait = education_learning_4
	trait = devoted 	# There is no evidence he was a monk per se, but no evidence he wasn't; an event will let player choose
	trait = zealous 	# Considered himself a staunch defender of orthodoxy against heresy, even if his opinions were controversial
	trait = patient 	# Worked on his thing for many years?
	trait = gregarious 	# Teacher
	trait = stubborn 	# Held controversial stances even in his own time
	trait = intellect_good_2
	trait = theologian
	trait = lifestyle_poet  	# Was also a competent poet
	815.11.5 = { # Known date according to Wikipedia, however Stanford Encyclopedia of Philosophy says his birth is unknown
		birth = yes
		give_nickname = nick_RICE_scottus
		effect = {			
			learn_language = language_greek 	# Unusually fluent and gifted in Greek, famous translator
			learn_language = language_latin 	# Also translated things into Latin
			learn_language = language_french 	# Probably knew some since he lived most his professional life in France
			add_character_flag = do_not_generate_starting_family
		}
	}
	850.1.1 = {		
		effect = { 
			set_relation_friend = {
				target = character:RICE_irish_004
				reason = friend_scholarly_matters
			}	
			set_relation_friend = {
				target = character:RICE_irish_005
				reason = friend_scholarly_matters
			}	
			# He was employed and patronized by Charles the Bald throughout his life, starting in the 850s
			set_relation_friend = {
				target = character:90104
				reason = RICE_friend_reason_placeholder
			}	
		}
	}
	877.7.8 = { # Supposed date, but this is uncertain per the sources
		death = yes
	}
}
