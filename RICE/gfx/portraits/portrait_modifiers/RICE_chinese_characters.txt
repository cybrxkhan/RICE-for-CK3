# We can use these modifiers for adding specific headgear to historical and vanity characters


# For some reason the game refuses to accept Lu You having a smaller beard, so here we are
RICE_chinese_lu_you_appearance = {

    usage = game
    selection_behavior = max
    priority = 10

    RICE_chinese_lu_you_hair_and_beard = {
        dna_modifiers = {
            accessory = {
                mode = add
                gene = beards
                template = tgp_chinese_beards
                accessory = m_beard_tgp_chinese_06
            }
        }
        weight = {
            base = 0
            modifier = {
                add = 333
				exists = this
				exists = character:han_11751
				this = character:han_11751
            }
        }
    }

}
