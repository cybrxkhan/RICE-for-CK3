RICE_upulvan = {
	type = character

	text = {
		trigger = {
			OR = {
				religion = { rice_religion_trigger_hinduism_religion = yes }
				culture = { has_cultural_pillar = heritage_dravidian }
			}
		}
		localization_key = RICE_upulvan_perumal_loc
	}

	text = {
		trigger = {
			OR = {
				religion = { rice_religion_trigger_buddhism_religion = yes }
				religion = { rice_religion_trigger_taoism_religion = yes }
				religion = { rice_religion_trigger_bon_religion = yes }
				religion = { rice_religion_trigger_vedda_religion = yes }
				culture = { has_cultural_pillar = heritage_indo_aryan }
			}
		}
		localization_key = RICE_upulvan_loc
		fallback = yes
	}

}
