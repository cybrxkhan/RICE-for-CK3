

RICE_mideast_spirit = {
	type = character

	text = {
		trigger = {
			OR = {
				faith = faith:south_mesopotamian_pagan
				faith = faith:harranian_pagan
				culture = { has_cultural_pillar = heritage_syriac }
			}
		}
		localization_key = RICE_mideast_deities.shida
	}

	text = {
		trigger = {
			OR = {
				religion = religion:zoroastrian_religion
                religion = religion:yazidi_religion
				culture = { has_cultural_pillar = heritage_iranian }
				culture = { has_cultural_pillar = heritage_elamite }
			}
		}
		localization_key = RICE_mideast_deities.pari
	}

	text = {
		trigger = {
			OR = {
				faith = faith:arabic_pagan
				faith = faith:nabataean_pagan
				religion = religion:islam_religion
				culture = { has_cultural_pillar = heritage_arabic }
			}
		}
		localization_key = RICE_mideast_deities.jinn
	}

	text = {
		localization_key = RICE_mideast_deities.spirit
		fallback = yes
	}

}
