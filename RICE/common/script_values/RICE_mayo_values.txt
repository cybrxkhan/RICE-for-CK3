
# Since the vanilla activity costs are unimmersively inflated at higher levels
RICE_butter_interaction_cost = {
	value = 25
	# multiply = {
	# 	value = primary_title.tier
	# 	subtract = 1
	# 	min = 1
	# }
	# multiply = {
	# 	value = primary_title.tier
	# 	subtract = 1
	# 	min = 1
	# }

	if = {
		limit = {
			scope:recipient = {
				is_ruler = yes
				highest_held_title_tier >= tier_empire
			}
		}
		multiply = 2.25
	}
	else_if = {
		limit = {
			scope:recipient = {
				OR = {
					AND = {
						is_ruler = yes
						highest_held_title_tier >= tier_kingdom
					}
					any_spouse = {
						is_ruler = yes
						highest_held_title_tier >= tier_empire
					}
					any_close_family_member = {
						is_ruler = yes
						highest_held_title_tier >= tier_empire
					}
				}
			}
		}
		multiply = 2
	}
	else_if = {
		limit = {
			scope:recipient = {
				OR = {
					AND = {
						is_ruler = yes
						highest_held_title_tier >= tier_duchy
					}
					any_spouse = {
						is_ruler = yes
						highest_held_title_tier >= tier_kingdom
					}
					any_close_family_member = {
						is_ruler = yes
						highest_held_title_tier >= tier_kingdom
					}
				}
			}
		}
		multiply = 1.75
	}
	else_if = {
		limit = {
			scope:recipient = {
				OR = {
					AND = {
						is_ruler = yes
						highest_held_title_tier >= tier_county
					}
					any_spouse = {
						is_ruler = yes
						highest_held_title_tier >= tier_duchy
					}
					any_close_family_member = {
						is_ruler = yes
						highest_held_title_tier >= tier_duchy
					}
				}
			}
		}
		multiply = 1.5
	}
	else_if = {
		limit = {
			scope:recipient = {
				OR = {
					AND = {
						is_ruler = yes
						highest_held_title_tier >= tier_barony
					}
					any_spouse = {
						is_ruler = yes
						highest_held_title_tier >= tier_county
					}
					any_close_family_member = {
						is_ruler = yes
						highest_held_title_tier >= tier_county
					}
				}
			}
		}
		multiply = 1.25
	}

	if = {
		limit = {
			government_has_flag = government_is_tribal
		}
		multiply = 0.5
	}
	# Make it a little easier for smaller realms
	if = {
		limit = {
			sub_realm_size <= 3
		}
		multiply = 0.75
	}
	min = 25
	max = 200
}
