activity_RICE_tarim_basin_khotan_procession = {
	is_shown = {
		highest_held_title_tier > tier_barony
		is_playable_character = yes
		OR = {
			any_held_title = {
				tier = tier_county
				any_county_province = {
					geographical_region = RICE_greater_khotan_area				
				}
			}
			completely_controls = title:c_khotan
			AND = {
				is_landless_adventurer = yes
				title:b_khotan = {
					squared_distance = {
						target = root.capital_province
						value <= squared_distance_large
					}
				}
			}
		}
		religion = religion:buddhism_religion
		NOT = { has_global_variable = RICE_tarim_basin_khotan_eight_protectors_abandoned }
		trigger_if = {
			limit = {
				is_ai = yes
			}
			is_at_war = no
			short_term_gold >= RICE_activity_cost
		}	
	}

	can_start = {
		religion = religion:buddhism_religion
		custom_description = {
			text = RICE_decision_khotan_procession_month_req
			current_month <= 4
		}
		custom_description = {
			text = RICE_decision_khotan_ruler_buddhist_req
			title:c_khotan.holder = {
				religion = religion:buddhism_religion
			}
		}
		trigger_if = {
			limit = { is_landless_adventurer = no }
			OR = {
				custom_description = {
					text = RICE_greater_khotan_area_geographical_req
					any_held_title = {
						tier = tier_county
						any_county_province = {
							geographical_region = RICE_greater_khotan_area				
						}
					}
				}
				completely_controls = title:c_khotan
			}
		}
		trigger_else = {
			custom_description = {
				text = RICE_tarim_basin_in_khotan_req
				domicile.domicile_location ?= {
					this = title:b_khotan.title_province
				}
			}
		}
		# Duh
		NOT = {
			is_at_war_with = title:c_khotan.holder
		}
	}

	can_start_showing_failures_only = {
		NOT = { is_activity_type_on_cooldown = activity_RICE_tarim_basin_khotan_procession }
		is_available_adult = yes

	is_valid = {
		NOT = {
			has_variable = activity_invalidated
		}

			is_playable_character = yes

		# 	}
		# 	has_attending_activity_guests = yes
		# }
		# Converting mid-activity is a big no-no.
		scope:activity = {
			OR = {
				NOT = { has_variable = starting_faith }
				# We could check whether the place you're going to is still a holy site in the new faith, but then that messes up all the people moving with you, so we just cancel the whole thing.
				scope:host.faith = var:starting_faith
			}
		}
	}

		# Host becomes unlanded
		if = {
			limit = {
				scope:host = { is_playable_character = no }

				activity_type = { save_scope_as = activity_type }
			}

				trigger_event = activity_system.0320
			}

		if = {
			limit = {
				scope:host = { is_incapable = yes }
			}
			scope:activity = {
				activity_type = { save_scope_as = activity_type }
			}

				trigger_event = activity_system.0330
			}
			every_attending_character = {
				limit = { NOT = { this = scope:host } }
				trigger_event = activity_system.0331
			}
		}
		# if = { # No one shows up

		# 		has_attending_activity_guests = no
		# 	}
		# 	scope:activity = {
		# 		activity_type = { save_scope_as = activity_type }
		# 		activity_location = { save_scope_as = location }
		# 	}
		# 	scope:host = { trigger_event = activity_system.0100 }
		# }
		# Host converted to a different faith
		# Could check to see if new faith allows for this kind of shenanigans, but it's easier to just do it broadly like this
		if = {
			limit = {
				NOT = { scope:host.faith = var:starting_faith }
				scope:host = { is_landed = yes } # Temporary solution to issue where landless adventurers get cancelled from doing the activity partway through
			}
			scope:activity = {
				activity_type = { save_scope_as = activity_type }
				activity_location = { save_scope_as = location }
			}
			every_attending_character = {
				trigger_event = RICEmisc.0010
			}
		}
	}
