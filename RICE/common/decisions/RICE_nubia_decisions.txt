	}

			add = 10
			has_trait = ambitious
		}

			current_date <= 1050.1.1
		}
		modifier = {
			add = 10
			# Happened roughly around the 11th/12th century IRL
			current_date > 1100.1.1
		}
		modifier = {
			add = 20
			# Happened roughly around the 11th/12th century IRL
