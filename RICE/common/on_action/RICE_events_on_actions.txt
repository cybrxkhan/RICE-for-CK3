RICE_events_yearly_maintenance_pulse = {

			limit = {

				}
			}
			random_list = {
				50 = {
					# Non-Buddhists less likely to care

						add = 15
						NOT = { religion = religion:buddhism_religion }

				}
				10 = {
					add_trait = RICE_mahavihara_trait
					if = {
