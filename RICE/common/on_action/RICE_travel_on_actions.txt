on_travel_plan_complete_RICE = {
	effect = {
		# Kill off any Merchant characters still left, because they always join the court for some reason
		current_travel_plan = {
			every_entourage_character = {
				limit = {
					has_variable = RICE_is_travel_option_merchant
				}
				# If they are a human player or a ruler for some reason, or got married or have family
				# Don't kill them, just remove the variable
				if = {
					limit = {		
						OR = {
							is_ai = no
							is_landed = yes
							is_married = yes
							has_no_particular_noble_roots_trigger = no
							has_character_flag = RICE_protected_merchant
						}		
					}
					remove_variable = RICE_is_travel_option_merchant
					if = {
						limit = {
							has_character_modifier = RICE_misc_focused_on_work
						}
						remove_character_modifier = RICE_misc_focused_on_work
					}
				}
				else = {
					death = {
						death_reason = death_vanished
					}
				}
			}
		}
	}
}

on_travel_plan_abort_RICE = {
	effect = {
		# Kill off any Merchant characters still left, because they always join the court for some reason
		current_travel_plan = {
			every_entourage_character = {
				limit = {
					has_variable = RICE_is_travel_option_merchant
				}
				# If they are a human player or a ruler for some reason, don't kill them, just remove the variable
				if = {
					limit = {		
						OR = {
							is_ai = no
							is_landed = yes
							is_married = yes
							has_no_particular_noble_roots_trigger = no
							has_character_flag = RICE_protected_merchant
							AND = {
								exists = dynasty
								dynasty = {
									any_dynasty_member = {
										count >= 1
									}
								}
							}
						}	
					}
					remove_variable = RICE_is_travel_option_merchant
				}
				else = {
					death = {
						death_reason = death_vanished
					}
				}
			}
		}
	}
}
