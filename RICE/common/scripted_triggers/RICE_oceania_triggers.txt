

RICE_culture_has_oceanic_heritage_pillar_trigger = {
	culture = {
		OR = {
			has_cultural_pillar = heritage_papuan
			has_cultural_pillar = heritage_micronesian
			has_cultural_pillar = heritage_melanesian
			has_cultural_pillar = heritage_polynesian
		}
	}
}

RICE_culture_has_oceanic_and_austronesian_heritage_pillar_trigger = {
	culture = {
		OR = {
			has_cultural_pillar = heritage_austronesian
			has_cultural_pillar = heritage_papuan
			has_cultural_pillar = heritage_micronesian
			has_cultural_pillar = heritage_melanesian
			has_cultural_pillar = heritage_polynesian
		}
	}
}

RICE_culture_has_non_papuan_oceanic_heritage_pillar_trigger = {
	culture = {
		OR = {
			has_cultural_pillar = heritage_micronesian
			has_cultural_pillar = heritage_melanesian
			has_cultural_pillar = heritage_polynesian
		}
	}
}

RICE_culture_has_australian_heritage_pillar_trigger = {
	culture = {
		OR = {
			has_cultural_pillar = heritage_arnhem_land
			has_cultural_pillar = heritage_kaurareg
		}
	}
}

RICE_culture_has_south_american_heritage_pillar_trigger = {
	culture = {
		OR = {
			has_cultural_pillar = heritage_andino
			has_cultural_pillar = heritage_indigenous_colombian
			has_cultural_pillar = heritage_mapuche
		}
	}
}

RICE_culture_has_greater_pacific_heritage_pillar_trigger = {
	OR = {
		RICE_culture_has_oceanic_heritage_pillar_trigger = yes
		RICE_culture_has_australian_heritage_pillar_trigger = yes
		RICE_culture_has_south_american_heritage_pillar_trigger = yes
	}
}

RICE_is_southeast_asian_or_oceanic_culture = {
	OR = {
		culture_has_southeast_asian_heritage_pillar_trigger = yes
		RICE_culture_has_oceanic_heritage_pillar_trigger = yes
	}
}

RICE_is_south_asian_southeast_asian_or_oceanic_culture = {
	OR = {
		culture_has_south_asian_heritage_pillar_trigger = yes
		RICE_is_southeast_asian_or_oceanic_culture = yes
	}
}

RICE_is_south_asian_southeast_asian_or_oceanic_culture_minus_polynesian = {
	OR = {
		culture_has_south_asian_heritage_pillar_trigger = yes
		culture_has_southeast_asian_heritage_pillar_trigger = yes
		culture = {
			OR = {
				has_cultural_pillar = heritage_papuan
				has_cultural_pillar = heritage_micronesian
				has_cultural_pillar = heritage_melanesian
			}
		}
	}
}

RICE_enjoys_kava_trigger = {
	OR = {
		has_trait = RICE_kava_drinker
		culture = { has_cultural_tradition = tradition_RICE_kava_culture }
		RICE_culture_has_oceanic_heritage_pillar_trigger = yes
		RICE_kava_religion_trigger = yes
	}
}

RICE_kava_religion_trigger = {
	OR = {
		religion = religion:micronesian_religion
		religion = religion:melanesian_religion
		religion = religion:papuan_religion
	}
}

RICE_enjoys_betel_nuts_trigger = {
	OR = {
		has_trait = betel_chewer
		RICE_is_south_asian_southeast_asian_or_oceanic_culture_minus_polynesian = yes
	}
}
