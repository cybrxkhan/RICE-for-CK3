RICE_normandy_can_be_granted_normandy_trigger = {
	# Should be a minor, low-tier ruler
	custom_description = {
		text = RICE_normandy_give_normandy_realm_req
		sub_realm_size <= RICE_normandy_give_normandy_realm_value
	}
	highest_held_title_tier <= tier_duchy
	# Should be Germanic
	culture = { has_cultural_pillar = heritage_north_germanic }
}


RICE_normandy_struggle_blocks_invasion_conquest_cb_trigger = {
	trigger_if = {
		limit = {
			scope:attacker = {
				any_character_struggle = {
					involvement = involved
					NOT = { is_struggle_type = RICE_normandy_struggle }
					NOT = { has_struggle_phase_parameter = invasion_conquest_war_cannot_be_declared }
				}
			}
			exists = scope:defender
		}
		scope:defender = {
			OR = {
				any_character_struggle = {
					involvement = involved
					NOT = { has_struggle_phase_parameter = invasion_conquest_war_cannot_be_declared }
				}
				any_character_struggle = {
					involvement = interloper
					NOT = { has_struggle_phase_parameter = invasion_conquest_war_cannot_be_declared }
				}
			}
		}
	}
	trigger_else = { always = yes }
}

RICE_normandy_donate_holy_site_req_trigger = {
	capital_county = {
		NOR = {
			has_county_modifier = RICE_normandy_built_chapel_county
			has_county_modifier = RICE_normandy_built_small_church_county
			has_county_modifier = RICE_normandy_built_small_monastery_county
			has_county_modifier = RICE_normandy_built_large_church_county
			has_county_modifier = RICE_normandy_built_large_monastery_county
		}
	}		
}
