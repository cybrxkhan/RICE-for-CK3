

RICE_baqt_nubian_christian_trigger = {

	culture = {
		OR = {
			has_cultural_pillar = heritage_east_african
			has_cultural_pillar = heritage_somalian
		}
	}
}

RICE_baqt_egyptian_coptic_trigger = {
	religion = religion:christianity_religion
	culture = {
		OR = {
			has_cultural_pillar = heritage_egyptian
			has_cultural_pillar = heritage_arabic
			has_cultural_pillar = heritage_berber
		}
	}
}

RICE_involved_in_baqt_trigger = {
	exists = situation:RICE_nubia_baqt_situation
	any_character_situation = {
		any_participant_group = {
			OR = {
				participant_group_type = RICE_nubia_baqt_nubian_christian_rulers
				participant_group_type = RICE_nubia_baqt_muslim_rulers
				participant_group_type = RICE_nubia_baqt_coptic_rulers
			}
		}
	}
}
