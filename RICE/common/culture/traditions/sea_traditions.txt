tradition_tiger_sons = {
	category = societal

	layers = {
		0 = martial
		1 = indian
		4 = knight.dds
	}

	is_shown = {
		OR ={
			this = culture:yi
			any_parent_culture_or_above = {
				this = culture:yi
			}
			this = culture:bai
			any_parent_culture_or_above = {
				this = culture:bai
			}
		}
	}
	can_pick = {
		culture_not_pacifistic_trigger = yes
	}

	parameters = {
		#unlock_maa_luojuzi = yes
		martial_education_more_valued = yes
		prowess_traits_more_valued = yes
		blademaster_trait_bonuses = yes
		craven_and_content_traits_looked_down_upon = yes
		knights_slightly_more_prone_to_injury = yes
		can_duel_rivals = yes
	}

	character_modifier = {
		knight_limit = 2
		negate_prowess_penalty_add = 5 # Wounded knights fight better!
	}

	cost = {
		prestige = {
			add = {
				value = tradition_base_cost
				desc = BASE
				format = "BASE_VALUE_FORMAT"
			}
			if = {
				limit = {
					NOR = {
						culture_pillar:ethos_bellicose = { is_in_list = traits }
						culture_pillar:ethos_stoic = { is_in_list = traits }
					}
				}
				add = {
					value = tradition_incompatible_ethos_penalty
					desc = not_bellicose_or_stoic_desc
				}
			}
			if = {
				limit = {
					NOT = {
						scope:character = {
							any_knight = {
								count >= 6
								prowess >= 12
							}
						}
					}
				}
				add = {
					value = tradition_unfulfilled_criteria_penalty
					desc = knights_with_prowess_desc
				}
			}

			multiply = tradition_replacement_cost_if_relevant
		}
	}
	ai_will_do = {
		value = 100
		if = {
			limit = {
				NOT = {
					culture_pillar:ethos_bellicose = { is_in_list = traits }
				}
			}
			multiply = 0.25
		}
	}
}



# Borrowed from Rajas of Asia
tradition_migratory = {
	category = realm

	layers = {
		0 = steward
		1 = mena
		4 = RICE_bush_branches.dds
	}

	can_pick = {
		NOT = { has_cultural_tradition = tradition_RICE_sea_nomads } # Sea Nomads is an "upgraded" version of Migratory
		OR = {
			custom_tooltip = {
				text = government_is_tribal_tt
				scope:character = {
					government_has_flag = government_is_tribal
				}
			}
			custom_tooltip = {
				text = nomadic_government_tt
				scope:character = {
					government_has_flag = government_is_nomadic
				}
			}
		}
	}

	parameters = {
		#enable_migration_cb = yes # RICE doesn't have this
		strong_traits_more_valued = yes
		more_likely_to_leave_court = yes
		unlock_voluntary_laampdom = yes
		much_more_likely_to_be_laamps = yes
	}
	county_modifier = {
		development_growth_factor = -0.25
		monthly_county_control_decline_factor = -0.25
	}
	character_modifier = {
		levy_size = 0.10
		movement_speed = 0.1
		character_travel_speed_mult = 0.05
		character_travel_safety_mult = 0.05
	}

	cost = {
		prestige = {
			add = {
				value = tradition_base_cost
				desc = BASE
				format = "BASE_VALUE_FORMAT"
			}
			if = {
				limit = {
					NOR = {
						culture_pillar:ethos_bellicose = { is_in_list = traits }
						culture_pillar:ethos_stoic = { is_in_list = traits }
						culture_pillar:ethos_communal = { is_in_list = traits }
					}
				}
				add = {
					value = tradition_incompatible_ethos_penalty
					desc = not_bellicose_stoic_or_communal_desc
				}
			}

			multiply = tradition_replacement_cost_if_relevant
		}
	}

	ai_will_do = {
		value = 0
	}
}
