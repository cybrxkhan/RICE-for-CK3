# SIWA

RICE_siwi = {
	trigger = {
		OR = {
			capital_county = {
				title:d_alexandria = {
					is_de_jure_liege_or_above_target = prev
				}
			}
			primary_title = title:d_alexandria
			capital_county = {
				title:d_al-wahat = {
					is_de_jure_liege_or_above_target = prev
				}
			}
			primary_title = title:d_al-wahat
		}
		culture = {
			has_cultural_pillar = heritage_berber
		}
		any_sub_realm_county = {
			culture = { has_cultural_pillar = heritage_berber }
			OR = {
				duchy = title:d_alexandria
				duchy = title:d_al-wahat
			}		
		}
	}
}

	trigger = {
		OR = {
			capital_county = {
				title:d_diyarmudar = {
					is_de_jure_liege_or_above_target = prev
				}
			}
			has_primary_title = title:d_diyarmudar
		}
		culture = {
			OR = {
				has_cultural_pillar = heritage_arabic
				has_cultural_pillar = heritage_byzantine
			}
		}
		OR = {
			faith.religion = { rice_religion_trigger_semitic_religion = yes }
			faith.religion = { rice_religion_trigger_hellenism_religion = yes }
			faith.religion = { rice_religion_trigger_kushitism_religion = yes }
			faith.religion = { rice_religion_trigger_amazigh_religion = yes }
		}
		any_sub_realm_county = {
			OR = {
				culture = { has_cultural_pillar = heritage_arabic }
				culture = { has_cultural_pillar = heritage_byzantine }
			}
			duchy = title:d_diyarmudar
		}	
	}
}
