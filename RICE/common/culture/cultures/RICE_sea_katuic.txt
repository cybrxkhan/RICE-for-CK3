# Kuy/Kuay
kuy = {
	color = { 128 158 188 }

	ethos = ethos_stoic
	heritage = heritage_mon_khmer
	language = language_mon_khmer # language_katuic in Rajas of Asia
	martial_custom = martial_custom_male_only
	head_determination = head_determination_domain

	traditions = {
		tradition_metal_craftsmanship 	# Cybr: In later centuries, the Kuy were famous ironworkers, which may have went as far back as the Angkorian period.
		tradition_jungle_dwellers 		# Cybr: They lived in the upland jungles in the Cambodia/Thailand/Laos border region.
		tradition_warrior_culture 		# Cybr: In later Cambodian sources, Kuay militia were renowned as fierce warriors.
		tradition_lords_of_the_elephant # Cybr: Wikipedia mentions them being known as skilled elephant trainers, but I couldn't find any scholarly sources corroborating this.
	}

	dlc_tradition = {
		trait = tradition_tgp_rice_cultivators
		requires_dlc_flag = all_under_heaven
		fallback = tradition_agrarian
	}

	name_list = name_list_khmer

	coa_gfx = { burman_group_coa_gfx }
	building_gfx = { southeast_asian_building_gfx }
	clothing_gfx = { southeast_asian_clothing_gfx }
	unit_gfx = { southeast_asian_unit_gfx indian_unit_gfx }
	house_coa_frame = house_frame_26
	house_coa_mask_offset = { 0.0 0.0 }
	house_coa_mask_scale = { 0.85 0.85 }

	ethnicities = {
		10 = asian_malay
	}
}
