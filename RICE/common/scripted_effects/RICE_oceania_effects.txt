RICE_oceania_sasi_ceremony_completed_log_entry_effect = {
	scope:activity = {
		add_activity_log_entry = {
			key = RICE_oceania_sasi_ceremony_completed_log
			tags = { completed }
			# this line below adds the entry to the Effects section of the conclusion UI
			show_in_conclusion = yes
			character = root
			if = {
				limit = {
					scope:host = { has_activity_intent = RICE_oceania_sasi_ceremony_conservation_intent }
				}
				scope:host = {
					add_prestige = 100
					stress_impact = {
						temperate = minor_stress_impact_loss
						patient = minor_stress_impact_loss
						diligent = minor_stress_impact_loss
					}
				}
			}
			else_if = {
				limit = {
					scope:host = { has_activity_intent = RICE_oceania_sasi_ceremony_ritual_intent }
				}
				scope:host = {
					add_piety = 100
					stress_impact = {
						zealous = minor_stress_impact_loss
						just = minor_stress_impact_loss
						humble = minor_stress_impact_loss
					}
				}
			}
			if = {
				limit = {
					scope:activity.activity_location = {
						county = {
							development_level >= good_development_level
						}
					}					
				}
				scope:host = {
					add_legitimacy = medium_legitimacy_gain
				}
			}
			else_if = {
				limit = {
