

RICE_venerate_ancestors_completed_log_entry_effect = {
	scope:activity = {
		add_activity_log_entry = {
			key = RICE_venerate_ancestors_completed_log
			tags = { completed }
			# this line below adds the entry to the Effects section of the conclusion UI
			show_in_conclusion = yes
			character = root
			custom_tooltip = RICE_activity_result_gained_piety
			custom_tooltip = RICE_activity_result_gained_prestige
			custom_tooltip = RICE_activity_result_gained_dynasty_prestige
			show_as_tooltip = {
				every_attending_character = {
					limit = { NOT = { this = scope:host } }
					custom = EVERY_ACTIVITY_PARTICIPANT_EFFECT
					custom_tooltip = RICE_activity_result_gained_piety
					custom_tooltip = RICE_activity_result_gained_prestige
				}
			}
			scope:host = {
				if = {
					limit = {
						scope:host.faith = { has_doctrine = rice_west_papuan_ancestor_focus_gosimo }
					}
					add_legitimacy = minor_legitimacy_gain
					every_vassal = {
						custom = custom.every_vassal
						add_opinion = {
							target = scope:actor
							modifier = pious_opinion
							opinion = 15
						}
					}
				}
				else_if = {
					limit = {
						scope:host.faith = { has_doctrine = rice_west_papuan_ancestor_focus_wor }
					}
					add_character_modifier = {
						modifier = RICE_oceania_ancestor_bonus_naval
						years = 10
					}
				}
				else_if = {
					limit = {
						scope:host.faith = { has_doctrine = rice_west_papuan_ancestor_focus_wuon }
					}
					add_character_modifier = {
						modifier = RICE_oceania_ancestor_bonus_mystical
						years = 10
					}
				}
				else_if = {
					limit = {
						scope:host.faith = { has_doctrine = rice_west_papuan_ancestor_focus_matutuo }
					}
					scope:activity.activity_location.county = {
						add_county_modifier = {
							modifier = RICE_oceania_ancestor_bonus_prosperity
							years = 10
						}
					}
				}
			}
		}
	}
}
