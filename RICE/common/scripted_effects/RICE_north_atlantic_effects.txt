RICE_setup_north_atlantic_after_game_start_effect = {
	# Guarantee it for at least two rulers
	random_ruler = {
		limit = {
			is_ai = yes
			exists = house
			NOT = { house = { has_house_modifier = RICE_north_atlantic_greenland_family_estate } }
			exists = struggle:RICE_greenland_struggle
			any_character_struggle = {
				involvement = involved
				is_struggle_type = RICE_greenland_struggle
			}
		}
		weight = {
			base = 100

			modifier = {
				add = 15
				highest_held_title_tier > tier_county
			}

			modifier = {
				add = 15
				highest_held_title_tier > tier_duchy
			}

			modifier = {
				add = 10
				stewardship > decent_skill_rating
			}

			modifier = {
				add = 10
				OR = {
					has_trait = diligent
					has_trait = ambitious
					has_trait = greedy
				}
			}

		}
		house = {
			add_house_modifier = {
				modifier = RICE_north_atlantic_greenland_family_estate
			}
		}
	}
	random_ruler = {
		limit = {
			is_ai = yes
			exists = house
			NOT = { house = { has_house_modifier = RICE_north_atlantic_greenland_family_estate } }
			exists = struggle:RICE_greenland_struggle
			any_character_struggle = {
				involvement = involved
				is_struggle_type = RICE_greenland_struggle
			}
		}
		weight = {
			base = 100

			modifier = {
				add = 5
				highest_held_title_tier > tier_county
			}

			modifier = {
				add = 5
				highest_held_title_tier > tier_duchy
			}

			modifier = {
				add = 10
				stewardship > decent_skill_rating
			}

			modifier = {
				add = 10
				OR = {
					has_trait = diligent
					has_trait = ambitious
					has_trait = greedy
				}
			}

		}
		house = {
			add_house_modifier = {
				modifier = RICE_north_atlantic_greenland_family_estate
			}
		}
	}
	if = {
		limit = {
			OR = {
				exists = struggle:RICE_greenland_struggle
				has_global_variable = RICE_greenland_ending_integration
				has_global_variable = RICE_greenland_ending_indigenous
			}			
		}
		every_ruler = {
			limit = {
				is_ai = yes
				exists = house
				NOT = { house = { has_house_modifier = RICE_north_atlantic_greenland_family_estate } }
				any_character_struggle = {
					involvement = involved
					is_struggle_type = RICE_greenland_struggle
				}				
			}
			random = {
				chance = 35
				modifier = {
					add = 5
					highest_held_title_tier > tier_county
				}
	
				# modifier = {
				# 	add = 5
				# 	highest_held_title_tier > tier_duchy
				# }
	
				modifier = {
					add = 5
					stewardship > decent_skill_rating
				}
	
				modifier = {
					add = 5
					OR = {
						has_trait = diligent
						has_trait = ambitious
						has_trait = greedy
					}
				}
				house = {
					add_house_modifier = {
						modifier = RICE_north_atlantic_greenland_family_estate
					}
				}
			}
		}
	}
}

			is_married = yes

		}
		every_spouse = {
			set_character_culture_icelandic = yes
		}
	}
	if = {
		limit = {
			any_close_family_member = {
				is_landed = no
			}
		}
		every_close_family_member = {
			custom = all_family_members
			limit = {
				is_landed = no
			}

			hidden_effect = {

RICE_north_atlantic_incorporate_inuit_ideas_effect = {
	add_character_modifier = {
		modifier = RICE_north_atlantic_inuit_alliance
	}
	add_prestige_level = -1
	set_global_variable = {
		name = RICE_greenland_adopted_arctic_adaptations
		value = yes
	}
	change_global_variable = {
		name = RICE_greenland_networks
		add = 100
	}
	change_global_variable = {
		name = RICE_greenland_resilience
		add = 200
	}
	if = {
		limit = {
			root.culture = { rice_culture_trigger_greenlandic = yes }
			culture:greenlandic = {
				NOT = { has_cultural_tradition = tradition_RICE_arctic_adaptations }
			}
		}
		custom_tooltip = RICE_north_atlantic_incorporate_inuit_ideas_greenlandic_effect
		hidden_effect = {
			culture:greenlandic = {
				add_culture_tradition = tradition_RICE_arctic_adaptations
				remove_variable = tradition_cooldown
				change_cultural_acceptance = {
					target = culture:inuit
					value = 25
					desc = RICE_cultural_acceptance_sharing_ideas_innovations
				}
			}	
		}	
	}
	else_if = {
		limit = {
			NOT = { root.culture = { rice_culture_trigger_greenlandic = yes } }
			root.culture = {
				NOT = { has_cultural_tradition = tradition_RICE_arctic_adaptations }
			}
		}
		custom_tooltip = RICE_north_atlantic_incorporate_inuit_ideas_your_culture_effect
		hidden_effect = {	
			root.culture = {			
				add_culture_tradition = tradition_RICE_arctic_adaptations
				remove_variable = tradition_cooldown
				change_cultural_acceptance = {
					target = culture:inuit
					value = 25
					desc = RICE_cultural_acceptance_sharing_ideas_innovations
				}
			}			
		}
	}
	if = {
		limit = {
			exists = struggle:RICE_greenland_struggle
		}
		struggle:RICE_greenland_struggle = {
			activate_struggle_catalyst = catalyst_RICE_incorporate_inuit_ideas
		}
	}
	hidden_effect = {
		RICE_greenland_learn_dog_sleds_effect = yes
		RICE_greenland_learn_toggling_harpoons_effect = yes
		RICE_greenland_learn_arctic_boats_effect = yes
	}
}

# Add cultural acceptance between Involved cultures
RICE_greenland_struggle_integration_cultural_acceptance_effect = {
	hidden_effect = {
		while = {
			limit = {
				any_in_global_list = {
					variable = RICE_greenland_struggle_ending_culture_list
					NOT = { has_variable = RICE_greenland_struggle_integration_flag }
				}
			}
			ordered_in_global_list = {
				variable = RICE_greenland_struggle_ending_culture_list
				limit = {
					NOT = { has_variable = RICE_greenland_struggle_integration_flag }
				}
				order_by = culture_number_of_counties
				save_scope_as = culture_scope
				set_variable = RICE_greenland_struggle_integration_flag # variable ensures acceptance is only gained once per couple
			}
			every_in_global_list = {
				variable = RICE_greenland_struggle_ending_culture_list
				limit = {
					NOR = {
						has_variable = RICE_greenland_struggle_integration_flag
						this = scope:culture_scope
					}
				}
				change_cultural_acceptance = {
					target = scope:culture_scope
					value = fp2_struggle_conciliation_culture_acceptance_value #TODO_CD_FP2 (JP) check if this is enough
					desc = RICE_greenland_struggle_integration_acceptance_tt
				}
			}
		}
		every_in_global_list = {
			variable = RICE_greenland_struggle_ending_culture_list
			remove_variable = RICE_greenland_struggle_integration_flag
		}
	}
}
