make_moksha_culture_effect = {
	#Convert your, and your whole family's, culture
	set_culture = culture:moksha
	if = {
		limit = {
			any_spouse = {
				is_landed = no				
				culture = { has_cultural_pillar = heritage_volga_finnic }
			}
		}
		every_spouse = {
			limit = {
				is_landed = no				
				culture = { has_cultural_pillar = heritage_volga_finnic }
			}
			set_culture = culture:moksha
		}
	}
	if = {
		limit = {
			any_close_family_member = {
				is_landed = no				
				culture = { has_cultural_pillar = heritage_volga_finnic }
			}
		}
		every_close_family_member = {
			custom = all_family_members
			limit = {
				is_landed = no				
				culture = { has_cultural_pillar = heritage_volga_finnic }
			}
			set_culture = culture:moksha
		}
	}

	#Convert appropriate vassals, and their family
	if = {
		limit = {
			any_vassal = {			
				culture = { has_cultural_pillar = heritage_volga_finnic }
			}
		}
		every_vassal = {
			limit = {
				culture = { has_cultural_pillar = heritage_volga_finnic }
			}
			set_culture = culture:moksha
			hidden_effect = {
				if = {
					limit = {
						any_spouse = {
							is_landed = no				
							culture = { has_cultural_pillar = heritage_volga_finnic }
						}
					}
					every_spouse = {
						limit = {
							is_landed = no				
							culture = { has_cultural_pillar = heritage_volga_finnic }
						}
						set_culture = culture:moksha
					}
				}
				if = {
					limit = {
						any_close_family_member = {
							is_landed = no				
							culture = { has_cultural_pillar = heritage_volga_finnic }
						}
					}
					every_close_family_member = {
						custom = all_family_members
						limit = {
							is_landed = no				
							culture = { has_cultural_pillar = heritage_volga_finnic }
						}
						set_culture = culture:moksha
					}
				}
			}
		}
	}

	#Convert appropriate courtiers
	if = {
		limit = {
			any_courtier_or_guest = {			
				culture = { has_cultural_pillar = heritage_volga_finnic }
			}
		}
		every_courtier_or_guest = {
			limit = {
				culture = { has_cultural_pillar = heritage_volga_finnic }
			}
			set_culture = culture:moksha
			hidden_effect = {
				if = {
					limit = {
						any_spouse = {
							is_landed = no				
							culture = { has_cultural_pillar = heritage_volga_finnic }
						}
					}
					every_spouse = {
						limit = {
							is_landed = no				
							culture = { has_cultural_pillar = heritage_volga_finnic }
						}
						set_culture = culture:moksha
					}
				}
				if = {
					limit = {
						any_close_family_member = {
							is_landed = no				
							culture = { has_cultural_pillar = heritage_volga_finnic }
						}
					}

RICE_mordvin_extract_beekeeper_tribute_effect = {

	if = {
		limit = {
			has_trait = greedy
		}
		stress_impact = {
			greedy = minor_stress_impact_loss
		}
	}
	else = {
		stress_impact = {
			compassionate = minor_stress_impact_gain
			generous = minor_stress_impact_gain
			just = minor_stress_impact_gain
		}
	}


	add_character_modifier = {
		modifier = RICE_mordvin_beekeeper_tribute_extracted
		years = 10
	}

	# Every county with the right tradition
	every_held_title = {
		custom = RICE_every_volga_beekeeper_county
		limit = {
			tier = tier_county
			culture = {
				has_cultural_tradition = tradition_RICE_volga_apiculture
			}
		}
		add_county_modifier = {
			modifier = RICE_mordvin_beekeeper_tribute_extracted_county
			years = 10
		}
	}

}
