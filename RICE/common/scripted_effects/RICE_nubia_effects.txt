	hidden_effect = {

			limit = {
				any_county_in_region = {
					region = RICE_north_africa_and_the_middle_east
					NOT = { culture = { has_same_culture_heritage = root.culture } }
				}				
			}
			random_county_in_region = {
				region = RICE_north_africa_and_the_middle_east
				limit = {
					NOT = { culture = { has_same_culture_heritage = root.culture } }
				}
				# Weight up according to development, so will tend to pick more prosperous counties
				weight = {
					base = 1
					modifier = {
						add = {
							value = development_level
							multiply = 1.5
						}
					}
					# Be more likely to select Arabic, Coptic, or Nubian cultures
					modifier = {
						OR = {
							culture = { has_cultural_pillar = heritage_arabic }
							culture = { has_cultural_pillar = heritage_egyptian }
							culture = culture:nubian
							culture = {
								any_parent_culture_or_above = {
									this = culture:nubian
								}
							}
						}
						add = 25
					}
				}
				save_scope_as = county_different
			}
		}
		# If one can't be found, just pick something random
		else = {
			random_county_in_region = {
				region = RICE_north_africa_and_the_middle_east
				# Weight up according to development, so will tend to pick more prosperous counties
				weight = {
					base = 1
					modifier = {
						add = {
							value = development_level
							multiply = 1.5
						}
					}
				}
				save_scope_as = county_different
			}		
		}
		create_character = {
			location = root.location
			template = RICE_random_gender_adjusted_character_template
			save_scope_as = random_different
			culture = scope:county_different.culture
			faith = scope:county_different.faith
		}
