make_south_arabian_culture_effect = {
	#Convert your, and your whole family's, culture
	set_character_culture_south_arabian = yes
	if = {
		limit = {
			knows_language = language_arabic
		}
		unlearn_language = language_arabic
	}
	if = {
		limit = {
			is_married = yes
			any_spouse = { is_landed = no }
		}
		every_spouse = {
			set_character_culture_south_arabian = yes
			if = {
				limit = {
					knows_language = language_arabic
				}
				unlearn_language = language_arabic
			}
		}
	}
	if = {
		limit = {
			any_close_family_member = {
				is_landed = no
			}
		}
		every_close_family_member = {
			custom = all_family_members
			limit = {
				is_landed = no
			}
			set_character_culture_south_arabian = yes
			if = {
				limit = {
					knows_language = language_arabic
				}
				unlearn_language = language_arabic
			}
		}
	}

	#Convert appropriate vassals, and their family
	if = {
		limit = {
			OR = {
				any_vassal = {
					is_ai = yes
				}
				any_courtier = {
					is_ai = yes
				}			
			}
		}
		every_vassal = {
			limit = {
				is_ai = yes
			}
			set_character_culture_south_arabian = yes
			unlearn_language = language_arabic
			hidden_effect = {
				if = {
					limit = { any_spouse = { is_landed = no } }
					every_spouse = {
						set_character_culture_south_arabian = yes
						if = {
							limit = {
								knows_language = language_arabic
							}
							unlearn_language = language_arabic
						}
					}
				}
				if = {
					limit = {
						any_close_family_member = {
							is_landed = no
						}
					}
					every_close_family_member = {
						limit = {
							is_landed = no
						}
						set_character_culture_south_arabian = yes
						if = {
							limit = {
								knows_language = language_arabic
							}
							unlearn_language = language_arabic
						}
					}
				}
			}
		}
		every_courtier = {
			limit = {
				is_ai = yes
			}
			set_character_culture_south_arabian = yes
			if = {
				limit = {
					knows_language = language_arabic
				}
				unlearn_language = language_arabic
			}
			hidden_effect = {
				if = {
					limit = { any_spouse = { is_landed = no } }
					every_spouse = {
						set_character_culture_south_arabian = yes
						if = {
							limit = {
								knows_language = language_arabic
							}
							unlearn_language = language_arabic
						}
					}
				}
				if = {
					limit = {
						any_close_family_member = {
							is_landed = no
						}
					}
					every_close_family_member = {
						limit = {
							is_landed = no
						}
						set_character_culture_south_arabian = yes
						if = {
							limit = {
								knows_language = language_arabic

RICE_add_pirate_den_effect = {
	capital_county = {
		add_county_modifier = {
			modifier = RICE_socotra_pirate_infestation
			years = 10
		}
	}
	if = {
		limit = {
			capital_province = {
				is_coastal = yes
			}
		}
		capital_province = {
			add_special_building = RICE_pirate_den_01
		}
	}
	else = {
		capital_county = {
			random_county_province = {
				limit = {
					is_coastal = yes
					has_holding = yes
				}
				add_special_building = RICE_pirate_den_01
			}		
		}
	}
}

form_socotra_kingdom_kingdom_effect = {
	dynasty = { add_dynasty_prestige = 1000 }
	#Create the title and make it primary
	create_title_and_vassal_change = {
		type = created
		save_scope_as = title_change
		add_claim_on_loss = no
	}
	title:k_socotra = {
		set_de_jure_liege_title = title:e_arabia
		change_title_holder = {
			holder = root
			change = scope:title_change
		}
	}
	resolve_title_and_vassal_change = scope:title_change
	set_primary_title_to = title:k_socotra
	
	set_realm_capital = title:c_socotra

	title:c_socotra = {
		add_county_modifier = {
			modifier = RICE_socotra_new_kingdom
			years = 100
		}
	}
	title:d_socotra = {
		set_de_jure_liege_title = title:k_socotra
	}
}

RICE_apply_ce_culture_changes = {
	# For CE

			exists = culture:mahri
		}
		culture:mahri = {
			if = {
				limit = { NOT = { has_cultural_tradition = tradition_RICE_ubiquity_of_poetry } }
				add_culture_tradition = tradition_RICE_ubiquity_of_poetry
				remove_variable = tradition_cooldown
			}
		}
	}	
	culture:south_arabian = {
		if = {
			limit = { NOT = { has_cultural_tradition = tradition_RICE_ubiquity_of_poetry } }
			add_culture_tradition = tradition_RICE_ubiquity_of_poetry
			remove_variable = tradition_cooldown
		}
	}
	# Sindhi Pirates
	culture:sindhi = {
		if = {
			limit = { NOT = { has_cultural_tradition = tradition_practiced_pirates } }
			add_culture_tradition = tradition_practiced_pirates
			remove_variable = tradition_cooldown
		}
	}
}
