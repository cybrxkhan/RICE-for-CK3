	}
	
}


# Siberian Courtier improves skills - tombola
siberia.0002 = {

	type = empty
	hidden = yes
	
	trigger = {
		# RICE_basic_is_available_adult_trigger = yes # Not needed because on_actions already handles this
		OR = {
			any_courtier = {
				RICE_siberia_siberian_wanderer_trigger = yes

				RICE_siberia_siberian_wanderer_trigger = yes

		}

				RICE_has_siberian_advanced_knowledge_trigger = no
				is_ai = yes			
			}
		}
	}

	immediate = {
		if = {
			limit = {
				any_vassal = {
					RICE_siberia_siberian_wanderer_trigger = yes
					RICE_basic_is_available_courtier_adult_trigger = yes
					is_ai = yes

				limit = {
					RICE_siberia_siberian_wanderer_trigger = yes
					RICE_basic_is_available_courtier_adult_trigger = yes
					is_ai = yes				
				}
				save_scope_as = scope_RICE_siberian_teacher
			}
		}
		else_if = {
			limit = {
				any_courtier = {
					RICE_siberia_siberian_wanderer_trigger = yes
					RICE_basic_is_available_courtier_adult_trigger = yes
					is_ai = yes			
				}

				limit = {
					RICE_siberia_siberian_wanderer_trigger = yes
					RICE_basic_is_available_courtier_adult_trigger = yes
					is_ai = yes				
				}
				save_scope_as = scope_RICE_siberian_teacher
			}		
		}
		# Choose who gets the lesson - slightly higher chance of another courtier getting it, but this can be changed
		random_list = {
			20 = {
				trigger = {
					RICE_has_siberian_knowledge_trigger = no
				}
				modifier = {
					add = 5
					learning >= decent_skill_rating
				}
				modifier = {
					add = 5
					has_trait = diligent
				}
				modifier = {
					add = 5
					has_trait = ambitious
				}
				modifier = {
					add = 5
					probably_intelligent_trigger = yes
				}
				add_character_flag = RICE_siberia_self_learner_basic
				trigger_event = siberia.0003
			}
			10 = {
				trigger = {
					RICE_has_siberian_knowledge_trigger = no
				}
				modifier = {
					add = 5
					learning >= decent_skill_rating
				}
				modifier = {
					add = 5

				}
				modifier = {

				}
				modifier = {

					probably_intelligent_trigger = yes
				}
				add_character_flag = RICE_siberia_self_learner_master
				trigger_event = siberia.0003
