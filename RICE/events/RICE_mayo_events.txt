# Ruaidrí Ua Conchobair / Rory O'Connor independent ruler notification event
# Minor rulers in Ireland must also choose between becoming a vassal or tributary
mayo.0016 = {
	type = character_event
	title = mayo.0016.t
	desc = mayo.0016.desc
	theme = crown
	override_background = { reference = RICE_background_irish_hills }

	left_portrait = {
		character = scope:english_ruler
		trigger = {
			exists = scope:english_ruler
		}
		animation = war_over_loss
	}

	right_portrait = {
		character = scope:rory_o_connor
		animation = war_over_win
	}

	immediate = {
		show_as_tooltip = {
			scope:rory_o_connor = {
				get_title = title:k_ireland
			}
		}
	}
	
	# Ok
	option = {
		name = mayo.0016.a
		trigger = {
			OR = {
				is_independent_ruler = no
				highest_held_title_tier > tier_duchy
				NOT = {
					any_held_title = {
						tier = tier_county
						any_county_province = {
							geographical_region = custom_ireland
						}
					}				
				}
			}
		}
	}
	
	# Remain independent
	option = {
		name = mayo.0016.b
		trigger = {
			is_independent_ruler = yes
			highest_held_title_tier <= tier_duchy
			any_held_title = {
				tier = tier_county
				any_county_province = {
					geographical_region = custom_ireland
				}
			}	
		}
		ai_chance = {
			factor = 50
			modifier = {
				has_trait = ambitious
				add = 25
			}
			modifier = {
				has_trait = greedy
				add = 15
			}
			modifier = {
				has_trait = content
				add = -15
			}
			modifier = {
				has_trait = craven
				add = -25
			}
		}
	}
	
	# Become a tributary
	option = {
		name = mayo.0016.c
		trigger = {
			is_independent_ruler = yes
			highest_held_title_tier <= tier_duchy
			any_held_title = {
				tier = tier_county
				any_county_province = {
					geographical_region = custom_ireland
				}
			}	
		}
		start_tributary_interaction_effect = {
			TRIBUTARY = root
			SUZERAIN = scope:rory_o_connor
		}
		ai_chance = {
			factor = 50
			modifier = {
				has_trait = calm
				add = 15
			}
			modifier = {
				has_trait = patient
				add = 15
			}
			modifier = {
				has_trait = greedy
				add = -15
			}
			modifier = {
				has_trait = ambitious
				add = -25
			}
		}
	}
	
	# Become a vassal
	option = {
		name = mayo.0016.d
		trigger = {
			is_independent_ruler = yes
			highest_held_title_tier <= tier_duchy
			any_held_title = {
				tier = tier_county
				any_county_province = {
					geographical_region = custom_ireland
				}
			}	
		}
		create_title_and_vassal_change = {
			type = swear_fealty
			save_scope_as = change
		}
		change_liege = {
			liege = scope:rory_o_connor
			change = scope:change
		}
		resolve_title_and_vassal_change = scope:change
		ai_chance = {
			factor = 50
			modifier = {
				has_trait = just
				add = 25
			}
			modifier = {
				has_trait = content
				add = 15
			}
			modifier = {
				has_trait = greedy
				add = -20
			}
			modifier = {
				has_trait = ambitious
				add = -40
			}
		}
	}

}

	type = character_event
	title = mayo.0053.t
	desc = mayo.0053.desc
	theme = RICE_theme_irish_monastery
	
	left_portrait = {
		character = root
		animation = thinking
	}
	
	right_portrait = {
		character = scope:saxon_monk
		animation = throne_room_bow_2
	}

	trigger = {
		has_global_variable = RICE_mayo_school_of_mayo_decision_english_happened
		religion = religion:christianity_religion
		capital_county = {
			any_county_province = {
				geographical_region = custom_ireland
			}
		}		
	}

	weight_multiplier = {
		base = 1
		modifier = {
			add = -0.85
			NOT = {
				has_global_variable = RICE_mayo_saxon_monk_spawn_timer
			}
		}
	}

	immediate = {
		if = {
			limit = {					
				any_county_in_region = {
					region = world_europe_west_britannia
					religion = religion:christianity_religion
				}
			}
			random_county_in_region = {
				region = world_europe_west_britannia
				limit = {
					religion = religion:christianity_religion
				}
				faith = {
					save_scope_as = county_faith
				}
			}
		}
		else = {
			random_list = {
				75 = {
					faith:catholic = {
						save_scope_as = county_faith
					}
				}
				25 = {
					faith:insular_celtic = {
						save_scope_as = county_faith
					}
				}
			}
		}
		if = {
			limit = {					
				any_county_in_region = {
					region = world_europe_west_britannia
					OR = {
						culture = { has_cultural_pillar = heritage_west_germanic }
						culture = { any_parent_culture_or_above = { this = culture:norman } }
					}
				}
			}
			random_county_in_region = {
				region = world_europe_west_britannia
				limit = {
					OR = {
						culture = { has_cultural_pillar = heritage_west_germanic }
						culture = { any_parent_culture_or_above = { this = culture:norman } }
					}
				}
				culture = {
					save_scope_as = county_culture
				}
			}
		}
		else = {
			culture:anglo_saxon = {
				save_scope_as = county_culture
			}
		}

		create_character = {
			location = root.capital_province
			template = RICE_random_monk_or_nun_template
			culture = scope:county_culture
			faith = scope:county_faith
			save_scope_as = saxon_monk
		}	
		hidden_effect = {
			scope:saxon_monk = {				
				add_trait = devoted
			}			
		}

	}
	
	# Ok
	option = {
		name = mayo.0053.a
		trigger = {
			OR = {
				gold >= 50
				is_ai = no
			}
		}
		remove_short_term_gold = 50
		add_courtier = scope:saxon_monk
	}
	
	# Ok
	option = {
		name = mayo.0053.b
		hidden_effect = {
			random = {
				chance = 90
				scope:saxon_monk = {
					death = {
						death_reason = death_vanished
					}
				}
			}
		}
	}
