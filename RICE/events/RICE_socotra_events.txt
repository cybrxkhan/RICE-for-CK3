			add = 0.2

		modifier = {
			add = -0.2
			has_trait = arbitrary
		}
	}

	option = { # Ok
		name = socotra.0060.a
		if = {
			limit = {
				NOR = {
					has_trait = vengeful
					has_trait = just
					has_trait = sadistic
				}

				base = minor_stress_impact_loss

				sadistic = miniscule_stress_impact_gain

	}
	option = { # Ok
