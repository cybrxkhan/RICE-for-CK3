# Fandaanano faith emerges
# Setup event
south_ethiopia.0001 = {
	type = character_event
	hidden = yes

	immediate = {
		save_scope_as = fandaanano_creator
		RICE_create_fandaanano_effect = yes
		add_prestige = 5000
		every_ruler = {
			limit = {
				OR = {
					this = scope:fandaanano_creator
					any_liege_or_above = { this = scope:fandaanano_creator }
					any_held_title = {
						tier = tier_county
						any_county_province = {
							geographical_region = RICE_south_ethiopia_region
						}
					}
				}
			}
			trigger_event = south_ethiopia.0002
		}
	}
}

# Fandaanano faith emerges
# Proper event
south_ethiopia.0002 = {
	type = character_event
	title = south_ethiopia.0002.t
	desc = south_ethiopia.0002.desc
	theme = RICE_theme_south_ethiopia_omotic_village

	right_portrait = {
		character = scope:fandaanano_creator
		animation = personality_zealous
	}

	immediate = {
		play_music_cue = "mx_cue_epic_sacral_moment"
		if = {
			limit = {
				root = scope:fandaanano_creator
			}
			add_prestige = 500
			dynasty ?= {
				add_dynasty_prestige = major_dynasty_prestige_value
			}
		}
		custom_tooltip = RICE_south_ethiopia_fandaanano_decision_effect_tooltip_1
		custom_tooltip = RICE_south_ethiopia_fandaanano_decision_effect_tooltip_2
	}

	# Ok
	option = {
		name = south_ethiopia.0002.a
		trigger = {
			faith = faith:fandaanano
		}
	}

	# Ok
	option = {
		name = south_ethiopia.0002.b
		trigger = {
			NOT = { faith = faith:fandaanano }
		}
		ai_chance = {
			base = 65
		}
	}

	# Convert
	option = {
		name = south_ethiopia.0002.c
		# Only certain types of AI rulers qualify
		trigger = {
			NOT = { faith = faith:fandaanano }
			OR = {
				is_ai = no
				capital_province = {
					geographical_region = RICE_south_ethiopia_region
				}		
			}
		}
		set_character_faith_with_conversion = faith:fandaanano
		ai_chance = {
			base = 35
			modifier = {
				NOR = {
					religion = religion:sidamic_religion
					religion = religion:waaqism_religion
					religion = religion:islam_religion
				}
				add = -40
			}
			modifier = {
				NOT = {
					capital_county = {
						faith = faith:fandaanano
					}
				}
				add = -20
			}
			modifier = {
				faith = faith:damot_pagan
				add = -20
			}
			modifier = {
				has_trait = zealous
				add = -30 
			}
			modifier = {
				has_trait = content
				add = -10 
			}
			modifier = {
				NOR = {
					culture = { has_cultural_pillar = heritage_somalian }
					culture = { has_cultural_pillar = heritage_east_african }
				}
				add = -30 
			}
		}
	}

}


# Aari/South Omotic character spawn
south_ethiopia.0010 = {
	type = character_event
	title = south_ethiopia.0010.t
	desc = south_ethiopia.0010.desc
	theme = RICE_theme_south_ethiopia_omotic_village

	left_portrait = root
	right_portrait = {
		character = scope:courtier
		animation = throne_room_bow_1
	}

	trigger = {
		any_held_county = {
			any_county_province = {
				geographical_region = RICE_south_ethiopia_region
			}
		}
		NOR = {
			culture = culture:aari
			culture = {
				any_parent_culture_or_above = {
					this = culture:aari
				}
			}
		}
	}

	weight_multiplier = {
		base = 1
		modifier = {
			add = 2.5
			culture = { has_cultural_tradition = tradition_RICE_great_rift_valley }
		}
	}

	immediate = {
		culture:aari = {
			save_scope_as = aari
		}
		create_character = {
			location = root.capital_province
			template = RICE_aari_generic_template
			save_scope_as = courtier			
		}
	}

	# Work with them
	option = {
		name = south_ethiopia.0010.a
		trigger = {
			OR = {
				gold >= 50
				is_ai = no
			}
		}
		remove_short_term_gold = 50
		add_courtier = scope:courtier
		if = {
			limit = {
				culture = { has_cultural_tradition = tradition_RICE_great_rift_valley }
			}
			capital_county = {
				add_county_modifier = {
					modifier = RICE_south_ethiopia_south_omotic_connections
					years = 20
				}
			}
		}
		else = {
			capital_county = {
				add_county_modifier = {
					modifier = RICE_south_ethiopia_south_omotic_connections
					years = 10
				}
			}
		}
	}
	
	# Don't work with them
	option = {
		name = south_ethiopia.0010.b
		hidden_effect = {
			random = {
				chance = 80
				scope:courtier = {
					death = {
						death_reason = death_vanished
					}
				}
			}
		}
	}

}

# Making the speech to boast to everyone

	type = activity_event
	title = south_ethiopia.0024.t

			}

				trigger = {
					scope:host = {
						var:RICE_south_ethiopia_cattle_counting_speech = flag:good
					}
				}
				desc = south_ethiopia.0024.desc.good
			}
			triggered_desc = {
				trigger = {
					scope:host = {
						var:RICE_south_ethiopia_cattle_counting_speech = flag:excellent
					}
				}
				desc = south_ethiopia.0024.desc.excellent
			}
		}
	}
	theme = RICE_theme_south_ethiopia_landscape
	
	right_portrait = {
		character = root
		animation = personality_bold
	}
	
	option = { # Ok
		name = south_ethiopia.0024.a
		if = {
			limit = {
				var:RICE_south_ethiopia_cattle_counting_speech = flag:excellent
			}
			scope:activity = {
				add_activity_log_entry = {
					key = RICE_south_ethiopia_cattle_counting_ceremony_speech_excellent_log
					tags = { good }

					show_in_conclusion = yes
					character = scope:host
					scope:host = {
						dynasty ?= {
							add_dynasty_prestige = RICE_cattle_counting_dynasty_prestige_excellent
						}
					}
				}
			}
		}
		else_if = {
			limit = {
				var:RICE_south_ethiopia_cattle_counting_speech = flag:good
			}
			scope:activity = {
				add_activity_log_entry = {
					key = RICE_south_ethiopia_cattle_counting_ceremony_speech_good_log
					tags = { good }
					# this line below adds the entry to the Effects section of the conclusion UI

					scope:host = {
						dynasty ?= {
							add_dynasty_prestige = RICE_cattle_counting_dynasty_prestige_good
						}
					}
				}
			}
		}
		else = {
			scope:activity = {
				add_activity_log_entry = {
					key = RICE_south_ethiopia_cattle_counting_ceremony_speech_fair_log

						dynasty ?= {
							add_dynasty_prestige = RICE_cattle_counting_dynasty_prestige_fair
						}
					}
				}
			}

	}

	after = {	
		if = {
			limit = { has_variable = RICE_south_ethiopia_cattle_counting_speech }
			remove_variable = RICE_south_ethiopia_cattle_counting_speech
		}

			limit = {

						category = special_type
						option = RICE_south_ethiopia_cattle_counting_ceremony_lika
					}
				}
			}

				id = south_ethiopia.0025
				days = { 2 4 }
			}
		}
		else = {
			trigger_event = {
				id = south_ethiopia.0026
				days = { 3 5 }
			}
		}
	}

}
