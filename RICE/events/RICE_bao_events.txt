# The Sonerien Du
bao.0001 = { # Two Black Musicians
	type = character_event
	title = bao.0001.t
	desc = bao.0001.desc
	theme = intrigue
	override_background = { reference = throne_room }

	right_portrait = {
		character = root
		animation = scheme
	}
	lower_left_portrait = {
		character = scope:vip_one
	}
	lower_center_portrait = {
		character = scope:vip_two
	}

	trigger = { 
	#	BAO_has_land_trigger = yes
  		NOT = { 
			root = { 
				has_character_flag = bao_sonerien_du_happened
			} 
		}
		has_trait = lifestyle_reveler
		culture = {
			OR = {
				rice_culture_trigger_breton = yes
				any_parent_culture_or_above = {
					rice_culture_trigger_breton = yes
				}
			}
		}
	}

	weight_multiplier = {
		base = 1
		modifier = {
			add = 0.2
			has_trait = just
		}
		modifier = {
			add = 0.2
			has_trait = wrathful
		}
		modifier = {
			add = -0.2
			has_trait = temperate
		}
		modifier = {
			add = -0.2
			has_trait = calm
		}
		modifier = {
			add = 2
			has_trait = gregarious
		}
		modifier = {
			add = 3
			has_trait = gluttonous
		}
		modifier = {
			add = 10
			has_trait = lifestyle_reveler
		}
	}

	immediate = { # Create the sonerien targets
		random_held_title = {
			limit = { tier = tier_county }
			save_scope_as = county
		}
		create_character = {
			employer = root
			age = { 20 32 }
			gender = male
			culture = root.culture
			faith = root.faith
			trait = education_diplomacy_3
			trait = beauty_good_3
			trait = gluttonous
			trait = gregarious
			trait = greedy
			trait = poet
			trait = lifestyle_reveler
			dynasty = generate
			save_scope_as = vip_one
		}
		create_character = {
			employer = root
			age = { 20 32 }
			gender = male
			culture = root.culture
			faith = root.faith
			martial = 12
			trait = education_diplomacy_4
			trait = physique_good_2
			trait = gluttonous
			trait = gregarious
			trait = patient
			trait = lifestyle_reveler
			dynasty = generate
			save_scope_as = vip_two
		}
  		add_character_flag = bao_sonerien_du_happened
	}

	option = { # Punish them!
		name = bao.0001.a
		stress_impact = {
			base = minor_stress_impact_gain
			just = major_stress_impact_gain
			calm = medium_stress_impact_gain
			compassionate = major_stress_impact_gain
		}
        add_gold = { 50 150 }
        scope:county = {
            add_county_modifier = {
                modifier = bao_sonerien_du_punished
                days = 1095
			}
		}
		scope:vip_one = {		
			death = {
				death_reason = death_execution
				killer = root
			}
		}
		scope:vip_two = {		
			death = {
				death_reason = death_execution
				killer = root
			}
		}
		if = {
			limit = {
				has_game_rule = RICE_historical_context_on
			}
			custom_tooltip = bao.0001.info
		}
	}

	option = { # Save them!
		name = bao.0001.b
		add_prestige = -100
		stress_impact = {
			callous = medium_stress_impact_gain
			arbitrary = medium_stress_impact_gain
			sadistic = medium_stress_impact_gain
		}
		scope:county = {
			add_county_modifier = {
				modifier = bao_sonerien_du_saved
				days = 1460
			}
		}
		hidden_effect = {
			scope:vip_one = {		
				death = {
					death_reason = death_disappearance
				}
			}
		}
		hidden_effect = {
			scope:vip_two = {
				death = {
					death_reason = death_disappearance	
				}
			}
		}
		if = {
			limit = {
				has_game_rule = RICE_historical_context_on
			}
			custom_tooltip = bao.0001.info
		}
	}

	option = { # Befriend Them!
		name = bao.0001.c
		add_prestige = -200
		if = {
			limit = {
				OR = {
					has_trait = shy
					has_trait = callous
					has_trait = paranoid
				}
			}
			stress_impact = {
				shy = minor_stress_impact_gain
				callous = minor_stress_impact_gain
				paranoid = minor_stress_impact_gain
			}
		}
		else = {
			stress_impact = {
				base = medium_stress_impact_loss
				compassionate = minor_stress_impact_loss
				gregarious = minor_stress_impact_loss
			}
		}
		add_character_modifier = {
			modifier = bao_sonerien_du_befriended
			years = 10
		}
		set_relation_friend = {
			reason = RICE_friend_reason_placeholder
			target = scope:vip_one
		}
		set_relation_friend = {
			reason = RICE_friend_reason_placeholder
			target = scope:vip_two
		}
		if = {
			limit = {
				has_game_rule = RICE_historical_context_on
			}
			custom_tooltip = bao.0001.info
		}
	}
}

			modifier = bao_run
			years = 10
		}
	}
	option = { # 2 Dance and chant !
		name = bao.0052.b
