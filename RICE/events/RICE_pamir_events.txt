	type = character_event

		character = scope:ali_ibn_asad

		animation = personality_content
	}
	
	immediate = {	
		character:RICE_pamir_006 = {
			save_scope_as = ali_ibn_asad
		}
		character:248022 = {
			save_scope_as = nasir_khusraw
		}
		# Have to do this for some reason even though it shows an error, as otherwise Nasir Khusraw will visit someone else's court even though he's your courtier
		add_courtier = scope:nasir_khusraw
		character:RICE_pamir_021 = {
			save_scope_as = nur_kayani
		}
		title:d_badakhshan = { #For loc.
			save_scope_as = badakhshan_duchy
		}
		if = {
			limit = {
				has_game_rule = RICE_historical_context_on
			}
			custom_description_no_bullet = {
				text = pamir.0009.info
			}
		}
	}
	
	option = { # Remain Sunni
		name = pamir.0009.a
		add_prestige = 100
		add_character_modifier = {
			modifier = RICE_pamir_ali_ibn_asad_unconverted
		}
	}	
	option = { # Convert to Ismailism
		name = pamir.0009.b
		add_piety = 100
		set_character_faith_with_conversion = faith:ismaili
		hidden_effect = {
			every_vassal = {
				set_character_faith_with_conversion = faith:ismaili
			}
			every_courtier = {
				set_character_faith_with_conversion = faith:ismaili
			}		
		}
		add_character_modifier = {
			modifier = RICE_pamir_ali_ibn_asad_converted
		}
	}
	after = {
		scope:nasir_khusraw = {
			return_to_court = yes
		}
	}
}
