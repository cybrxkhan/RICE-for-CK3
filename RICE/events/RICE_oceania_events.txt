oceania.0035 = {
	type = character_event
	hidden = yes

	# Make sure the character is still someone who can actually get a courtier
	trigger = {
		NOT = { has_game_rule = RICE_oceania_frequency_off }
		is_playable_character = yes
	}

	weight_multiplier = {
		base = 1
		modifier = {
			has_character_modifier = RICE_oceania_pacific_connections
			add = 1
		}
		modifier = {
			has_game_rule = RICE_oceania_frequency_uncommon
			add = 0.5
		}
		modifier = {
			has_game_rule = RICE_oceania_frequency_common
			add = 1
		}
	}	

	immediate = {
		random_list = {
			# Spawn Zone 1 characters: West Papua, West Micronesia
			110 = { trigger_event = oceania.0030 }
			# Spawn Zone 2 characters: Central Micronesia, Central Papua
			60 = { trigger_event = oceania.0031 }
			# Spawn Zone 3 characters: East Micronesia, East Papua, West Melanesia, Australia
			30 = { trigger_event = oceania.0032 }
			# Spawn Zone 4 characters: Polynesia, East Melanesia
			20 = {
				trigger = {
					has_game_rule = RICE_remote_oceania_on
				}
				trigger_event = oceania.0033
			}
			# Spawn Zone 5 characters: South America
			10 = {
				trigger = {
					has_game_rule = RICE_south_america_on
				}
				trigger_event = oceania.0034
			}
		}
	}
}
