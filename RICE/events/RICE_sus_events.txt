# Encounter Trans-Saharan merchants
sus.0001 = {
	type = character_event
	title = sus.0001.t
	desc = sus.0001.desc
	theme = travel_stewardship
	
	left_portrait = root
	right_portrait = scope:merchant

	trigger = {
		current_travel_plan = {
			any_entourage_character = {
				count >= 1 
				NOT = { this = root }
				has_trait = RICE_trans_saharan_merchant
			}
		}
	}

	immediate = {
		current_travel_plan = {
			random_entourage_character = {
				limit = {
					NOT = { this = root }
					has_trait = RICE_trans_saharan_merchant
				}
				save_scope_as = merchant
			}
		}		
	}

	option = { # OK
		name = sus.0001.a
		add_gold = { 10 50 }
		stress_impact = {
			greedy = minor_stress_impact_gain
		}

		if = {
			limit = {
				exists = var:RICE_trans_saharan_merchant_success_count
				OR = {
					var:RICE_trans_saharan_merchant_success_count >= 10
					dynasty ?= {
						has_dynasty_perk = RICE_trans_saharan_trade_legacy_5
					}
				}
				NOT = { has_trait = RICE_trans_saharan_merchant }
			}
			random = {
				chance = RICE_trans_saharan_merchant_trait_chance
				add_trait = RICE_trans_saharan_merchant
				custom_tooltip = RICE_trans_saharan_merchant_trait_gain_tooltip
			}
		}
		if = {
			limit = {
				has_trait = RICE_trans_saharan_merchant
			}
			add_trait_xp = {
				trait = RICE_trans_saharan_merchant
				value = { 1 5 } 
			}
		}

		hidden_effect = {
			if = {
				limit = {
					NOT = { exists = var:RICE_trans_saharan_merchant_success_count }
				}
				set_variable = {
					name = RICE_trans_saharan_merchant_success_count
					value = 1
				}
			}
			else = {
				change_variable = {
					name = RICE_trans_saharan_merchant_success_count
					add = 1
				}		
			}
		}

		if = {
			limit = {
				exists = var:RICE_trans_saharan_merchant_success_count
				# NOR = {
				# 	var:RICE_trans_saharan_merchant_success_count >= 10
				# 	dynasty ?= {
				# 		has_dynasty_perk = RICE_trans_saharan_trade_legacy_5
				# 	}
				# }
			}
			custom_tooltip = RICE_trans_saharan_number_of_successes_tooltip
		}
	}

	option = { # OK
		name = sus.0001.b
		duel = {
			skills = { diplomacy stewardship }
			value = decent_skill_rating
			# Success.
			30 = {
				compare_modifier = {
					value = scope:duel_value
					multiplier = 2.5
					min = -49
				}
				custom_tooltip = sus.0001.b.success
				show_as_tooltip = {
					if = {
						limit = {
							exists = var:RICE_trans_saharan_merchant_success_count
							OR = {
								var:RICE_trans_saharan_merchant_success_count >= 10
								dynasty ?= {
									has_dynasty_perk = RICE_trans_saharan_trade_legacy_5
								}
							}
							NOT = { has_trait = RICE_trans_saharan_merchant }
						}
						random = {
							chance = RICE_trans_saharan_merchant_trait_chance
							add_trait = RICE_trans_saharan_merchant
							custom_tooltip = RICE_trans_saharan_merchant_trait_gain_tooltip
						}
					}
					if = {
						limit = {
							has_trait = RICE_trans_saharan_merchant
						}
						add_trait_xp = {
							trait = RICE_trans_saharan_merchant
							value = { 1 5 } 
						}
					}
				}
				hidden_effect = {
					send_interface_toast = {
						type = event_toast_effect_good
						title = sus.0001.b.success
						left_icon = root
						add_gold = { 50 150 }
						if = {
							limit = {
								exists = var:RICE_trans_saharan_merchant_success_count
								OR = {
									var:RICE_trans_saharan_merchant_success_count >= 10
									dynasty ?= {
										has_dynasty_perk = RICE_trans_saharan_trade_legacy_5
									}
								}
								NOT = { has_trait = RICE_trans_saharan_merchant }
							}
							random = {
								chance = RICE_trans_saharan_merchant_trait_chance
								add_trait = RICE_trans_saharan_merchant
								custom_tooltip = RICE_trans_saharan_merchant_trait_gain_tooltip
							}
						}
						if = {
							limit = {
								has_trait = RICE_trans_saharan_merchant
							}
							add_trait_xp = {
								trait = RICE_trans_saharan_merchant
								value = { 1 5 } 
							}
						}
					}	
					if = {
						limit = {
							NOT = { exists = var:RICE_trans_saharan_merchant_success_count }
						}
						set_variable = {
							name = RICE_trans_saharan_merchant_success_count
							value = 1
						}
					}
					else = {
						change_variable = {
							name = RICE_trans_saharan_merchant_success_count
							add = 1
						}		
					}					
				}
			}
			# Failure.
			70 = {
				compare_modifier = {
					value = scope:duel_value
					multiplier = -2.5
					min = -49
				}
				custom_tooltip = sus.0001.b.failure
				hidden_effect = {
					send_interface_toast = {
						type = event_toast_effect_bad
						title = sus.0001.b.failure
						left_icon = root
						add_prestige = -100
					}					
				}
			}
		}
		if = {
			limit = {
				exists = var:RICE_trans_saharan_merchant_success_count
				# NOR = {
				# 	var:RICE_trans_saharan_merchant_success_count >= 10
				# 	dynasty ?= {
				# 		has_dynasty_perk = RICE_trans_saharan_trade_legacy_5
				# 	}
				# }
			}
			custom_tooltip = RICE_trans_saharan_number_of_successes_tooltip
		}		
	}

}

			geographical_region = world_africa

						}
						AND = {
							has_global_variable = RICE_game_started_on_TFE_start_date
							culture = { has_cultural_pillar = heritage_levantine }
						}
					}
				}
			}

						culture = { has_cultural_pillar = heritage_arabic }
